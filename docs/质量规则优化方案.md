# 数据质量管理模块优化方案

## 📊 当前系统分析

### 1. 主数据管理表单结构

根据分析，当前系统有以下核心表：

#### 数据标准与模型中心
- **std_field** - 字段标准库（字段定义、类型、校验规则）
- **std_view** - 视图定义（主数据视图）
- **std_view_entity** - 实体定义（主实体/子实体）
- **std_view_field** - 视图字段（字段属性、校验配置）
- **std_value_domain** - 值域管理（枚举值、参照数据）
- **std_encoding_rule** - 编码规则

#### 主数据实例
- **mst_type** - 主数据类型
- **mst_instance** - 主数据实例（实际业务数据）

#### 质量管理（当前）
- **qlt_rule** - 质量规则
- **qlt_check_task** - 检测任务
- **qlt_check_result** - 检测结果
- **qlt_issue** - 问题管理

### 2. 当前质量规则存在的问题

#### ❌ 规则配置不够完善
```java
// 当前字段
private String ruleConfig;  // 配置内容为TEXT，缺乏结构化定义
private BigDecimal threshold; // 只有阈值，缺少详细的检查参数
```

**问题：**
1. `ruleConfig` 字段缺乏结构化定义，难以配置复杂规则
2. 缺少具体的检查类型配置（如正则表达式、值域范围、关联检查等）
3. 没有规则模板功能，每次都要重新配置
4. 缺少规则测试功能，无法验证规则是否正确

#### ❌ 检查类型实现不完整
```java
// 当前实现
case "completeness":
    return fieldValue != null && !fieldValue.trim().isEmpty();
case "uniqueness":
    return true; // 简化为通过
case "accuracy":
    return checkAccuracy(fieldValue, rule); // TODO: 未实现
```

**问题：**
1. 完整性检查只实现了简单的非空检查
2. 唯一性检查未实现
3. 准确性检查未实现
4. 一致性检查未实现
5. 及时性检查未实现

#### ❌ 缺少规则与字段标准的关联
- 字段标准库（std_field）已有校验规则，但质量规则没有利用
- 值域管理（std_value_domain）未与质量规则关联
- 视图字段的校验配置（std_view_field）与质量规则脱节

```java
/**
 * 完整性检查
 */
private boolean checkCompleteness(Object[] row, QualityRule rule) {
    JSONObject config = JSON.parseObject(rule.getCheckConfig());
    String checkType = config.getString("checkType");
    Object fieldValue = row[1];

    switch (checkType) {
        case "not_null":
            // 非空检查
            boolean trimWhitespace = config.getBooleanValue("trimWhitespace", true);
            if (fieldValue == null) return false;
            if (trimWhitespace && fieldValue instanceof String) {
                return !((String) fieldValue).trim().isEmpty();
            }
            return true;

        case "not_empty":
            // 非空字符串
            if (fieldValue == null) return false;
            String strValue = fieldValue.toString();
            int minLength = config.getIntValue("minLength", 1);
            Integer maxLength = config.getInteger("maxLength");
            int length = strValue.length();
            if (length < minLength) return false;
            if (maxLength != null && length > maxLength) return false;
            return true;

        default:
            return true;
    }
}
```

```java
/**
 * 准确性检查
 */
private boolean checkAccuracy(Object[] row, QualityRule rule) {
    JSONObject config = JSON.parseObject(rule.getCheckConfig());
    String checkType = config.getString("checkType");
    Object fieldValue = row[1];

    if (fieldValue == null) {
        return config.getBooleanValue("allowNull", true);
    }

    switch (checkType) {
        case "regex":
            // 正则表达式检查
            String pattern = config.getString("pattern");
            String flags = config.getString("flags", "");
            return Pattern.compile(pattern, parseFlags(flags))
                .matcher(fieldValue.toString()).matches();

        case "domain":
            // 值域检查
            String domainCode = config.getString("domainCode");
            return checkValueInDomain(fieldValue, domainCode);

        case "range":
            // 范围检查
            return checkValueInRange(fieldValue, config);

        case "length":
            // 长度检查
            return checkValueLength(fieldValue, config);

        default:
            return true;
    }
}

/**
 * 值域检查
 */
private boolean checkValueInDomain(Object value, String domainCode) {
    // 查询值域项
    String sql = "SELECT COUNT(*) FROM std_value_domain_item " +
                "WHERE domain_code = ? AND item_value = ? AND status = 'active'";
    Query query = entityManager.createNativeQuery(sql);
    query.setParameter(1, domainCode);
    query.setParameter(2, value.toString());

    BigInteger count = (BigInteger) query.getSingleResult();
    return count.intValue() > 0;
}

/**
 * 范围检查
 */
private boolean checkValueInRange(Object value, JSONObject config) {
    BigDecimal numValue = new BigDecimal(value.toString());
    BigDecimal minValue = config.getBigDecimal("minValue");
    BigDecimal maxValue = config.getBigDecimal("maxValue");
    String minOperator = config.getString("minOperator", ">=");
    String maxOperator = config.getString("maxOperator", "<=");

    boolean minPass = minValue == null ||
        (">=".equals(minOperator) ? numValue.compareTo(minValue) >= 0 : numValue.compareTo(minValue) > 0);
    boolean maxPass = maxValue == null ||
        ("<=".equals(maxOperator) ? numValue.compareTo(maxValue) <= 0 : numValue.compareTo(maxValue) < 0);

    return minPass && maxPass;
}
```

```java
    Map<String, Object> result = qualityRuleService.testRule(dto);

/**
 * 测试规则
 */
public Map<String, Object> testRule(QualityRuleDto dto) {
    Map<String, Object> result = new HashMap<>();

    // 限制只检测100条
    String tableName = dto.getTableName();
    String fieldCode = dto.getFieldCode();
    String sql = String.format(
        "SELECT id, %s FROM %s WHERE status = 'active' LIMIT 100",
        fieldCode, tableName
    );

    Query query = entityManager.createNativeQuery(sql);
    List<Object[]> rows = query.getResultList();

    int passCount = 0;
    int failCount = 0;
    List<Map<String, Object>> failRecords = new ArrayList<>();

    for (Object[] row : rows) {
        boolean passed = executeRuleCheck(row, dto);
        if (passed) {
            passCount++;
        } else {
            failCount++;
            if (failRecords.size() < 10) { // 只显示前10条失败记录
                failRecords.add(Map.of(
                    "recordId", row[0],
                    "fieldValue", row[1] != null ? row[1].toString() : null,
                    "reason", generateIssueMessage(dto)
                ));
            }
        }
    }

    result.put("totalRecords", rows.size());
    result.put("passCount", passCount);
    result.put("failCount", failCount);
    result.put("passRate", rows.size() > 0 ? (passCount * 100.0 / rows.size()) : 100);
    result.put("failRecords", failRecords);

    return result;
}
```

```java
/**
```

```java
 */
public QualityRule generateFromFieldStandard(Long fieldStandardId) {
    StdFieldStandard fieldStandard = fieldStandardRepository.findById(fieldStandardId)

    rule.setRuleCode("AUTO_" + fieldStandard.getFieldCode());

    // 必填检查
    if (fieldStandard.getIsRequired()) {
        rule.setRuleType("completeness");
        rule.setCheckType("not_null");
        config.put("checkType", "not_null");
    }

    // 长度检查
    if (fieldStandard.getLength() != null) {
        rule.setRuleType("accuracy");
        rule.setCheckType("length");
        config.put("checkType", "length");
        config.put("maxLength", fieldStandard.getLength());
    }

    // 正则校验
    if (fieldStandard.getValidationRule() != null) {
        rule.setRuleType("accuracy");
        rule.setCheckType("regex");
        config.putAll(JSON.parseObject(fieldStandard.getValidationRule()));
    }

    rule.setCheckConfig(config.toJSONString());
    return rule;
}
```
